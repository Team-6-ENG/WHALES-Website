package com.events.hiddenevents;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Rectangle;
import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.mazegame.Main;
import java.util.Objects;

/** Represents hidden event of professor who doesn't stop talking Added 28/12/25 . */
public class RamblingProfessorEvent extends HiddenEvent {

  static String IMG = "characters/professor.png";
  static String FILE = "professorText.txt";
  private final Texture profTexture = new Texture(Gdx.files.internal(IMG));
  private final String[] speechList;
  private final Rectangle bounds;
  private final Label nameLabel;
  private float timeSinceTalk = 10;
  private String currentLine = "";
  private int speechIndex = 0;

  /**
   * Professor constructor.
   *
   * @param start random index of text to start on
   */
  public RamblingProfessorEvent(int start) {
    super(new Vector2(15, 15));
    speechIndex = start;
    bounds = new Rectangle(16, 17, 3, 3);
    nameLabel = new Label("Professor", Main.labelStyle);
    nameLabel.setFontScale(0.02f);
    nameLabel.setPosition(16, Main.WORLD_HEIGHT - 11);
    String text = Gdx.files.internal(FILE).readString();

    speechList = text.split("\n");
  }

  /**
   * Updates the line which the professor is saying.
   *
   * @param delta time in seconds since last frame call
   * @return String of speech
   */
  public String talk(float delta) {
    timeSinceTalk += delta;
    if (timeSinceTalk > 5) {
      timeSinceTalk = 0;
      currentLine = speechList[speechIndex];
      speechIndex++;
      if (speechIndex == speechList.length) {
        speechIndex = 0;
      }
    }

    return currentLine;
  }

  /**
   * Get professor bounds.
   *
   * @return Rectangle bounds representing area containing professor
   */
  public Rectangle getBounds() {
    return bounds;
  }

  /**
   * Draw professor to screen if in correct room.
   *
   * @param batch game spritebatch
   * @param location current location
   */
  public void render(SpriteBatch batch, String location) {
    if (Objects.equals(location, Main.professorRoom)) {
      batch.draw(profTexture, bounds.x, bounds.y, bounds.width, bounds.height);
      nameLabel.draw(batch, 1f);
    }
  }
}
