package com.events.hiddenevents;

import com.audio.AudioManager;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import com.badlogic.gdx.math.Vector2;
import com.badlogic.mazegame.Main;
import java.util.ArrayList;

/** Makes wind along the main path Created 16/12/2025. */
public class WindEvent extends HiddenEvent {

  static final String LEAF_IMG = "items/leaf.png";

  /** Each leave has an x, a y and rotation. */
  private final ArrayList<ArrayList<Float>> leaves = new ArrayList<>();

  private final float leafSpeed = 6f;
  Texture leafTexture = new Texture(Gdx.files.internal(LEAF_IMG));
  private int windy = 0;

  /** Create a new hidden event at the given position. */
  public WindEvent() {
    super(new Vector2(0.0f, 0.0f));
    deactivate();
  }

  /**
   * Check if wind should change direction when entering main path.
   *
   * @param change will change if equals 1
   * @param direction new direction for wind if change
   * @param room current location of player
   * @param previousRoom previous location of player
   */
  public void checkWindTrigger(
      int change, int direction, String room, String previousRoom, AudioManager audio) {

    if (change == 1 && room.contains("MainPath") && !previousRoom.contains("MainPath")) {
      windy = direction;

      if (!triggered && windy != 0) {
        trigger();
      }
      if (windy == 0) {
        leaves.clear();
        audio.stopWind();
        deactivate();
      }
    }
    if (room.contains("MainPath") && windy != 0) {
      audio.playWind();
    } else {
      audio.stopWind();
      deactivate();
    }
  }

  /**
   * Update the position of all leaves.
   *
   * @param newLeaf will add new leaf if equals 1
   * @param newLeafY height of new leaf if added
   * @param delta time in seconds since last call
   */
  public void updateLeaves(int newLeaf, float newLeafY, float delta) {
    if (windy != 0) {

      ArrayList<ArrayList<Float>> outOfBoundsLeaves = new ArrayList<>();
      for (ArrayList<Float> leaf : leaves) {
        // Update x of leaf
        leaf.set(0, leaf.get(0) + (leafSpeed * delta * windy));
        // update rotation of leaf
        leaf.set(2, leaf.get(2) + (leafSpeed * delta * 10));
        if (leaf.get(0) > (((float) Main.WORLD_WIDTH / 2) + ((float) (Main.WORLD_HEIGHT) / 2) - 2)
            || leaf.get(0) < (((float) Main.WORLD_WIDTH / 2) - ((float) (Main.WORLD_HEIGHT) / 2))) {
          outOfBoundsLeaves.add(leaf);
        }
      }
      for (ArrayList<Float> leaf : outOfBoundsLeaves) {
        leaves.remove(leaf);
      }

      if (newLeaf == 1) {
        ArrayList<Float> pos = new ArrayList<>();
        pos.add(
            (float) ((Main.WORLD_WIDTH / 2) + (Main.WORLD_HEIGHT * windy * -1 / 2))
                - (windy == -1 ? 2 : 0));
        pos.add(newLeafY);
        pos.add(newLeafY * 10);
        leaves.add(pos);
      }
    }
  }

  /**
   * Render the leaves to the screen.
   *
   * @param batch The SpriteBatch to render to
   */
  public void render(SpriteBatch batch, String room) {
    if (room.contains("MainPath")) {
      for (ArrayList<Float> leaf : leaves) {

        batch.draw(
            new TextureRegion(leafTexture),
            leaf.get(0),
            leaf.get(1),
            1.5f,
            1.5f,
            1.5f,
            1,
            1,
            1,
            leaf.get(2));
      }
    }
  }

  public int getWindy() {
    return windy;
  }

  public int getNumberOfLeaves() {
    return leaves.size();
  }

  public ArrayList<ArrayList<Float>> getLeaves() {
    return leaves;
  }
}
