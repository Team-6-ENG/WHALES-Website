package com.events;

import com.badlogic.gdx.graphics.g2d.Sprite;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import com.badlogic.gdx.math.Rectangle;
import com.badlogic.mazegame.Main;
import com.characters.Animator;
import com.characters.Player;
import java.util.HashMap;

/** Represents player eating frog. */
public class Frog extends Animator {

  Sprite sprite;
  TextureRegion frame;
  private float stateTime;
  private float eatTime;
  private boolean hasPlayer;
  private int escapeCounter;
  private float punched;
  private boolean escaping;

  /** Loads sprite sheet. */
  public Frog() {
    super("characters/frog.png", 4, 3);

    HashMap<String, Integer[]> animationInfo = new HashMap<>();
    // HashMap<String, Integer[]> animationInfo:
    //      key - Name of animation
    //      Value - Array representing row of animation on sprite sheet and index of start and end
    // frames
    animationInfo.put("idle", new Integer[] {0, 0, 2});
    animationInfo.put("eyes", new Integer[] {1, 0, 2});
    animationInfo.put("croak", new Integer[] {2, 0, 4});

    generateAnimation(animationInfo, 0.6f);

    animations.get("croak").setFrameDuration(0.2f);
    sprite = new Sprite(animations.get("idle").getKeyFrame(0, true));

    sprite.setPosition(20, 2);
  }

  /**
   * Update animations and frog state.
   *
   * @param delta time in seconds since last frame call.
   */
  public void update(float delta) {
    if (!hasPlayer) {
      frame = animations.get("idle").getKeyFrame(stateTime, true);
    } else if (eatTime < 0.8f) {
      frame = animations.get("croak").getKeyFrame(eatTime, true);

    } else {
      frame = animations.get("eyes").getKeyFrame(eatTime, true);
    }

    sprite.setSize(10f, 8);
    if (punched > 1) {
      sprite.setSize(10 * punched, 8 * punched);
      punched -= delta;
    }
    sprite.setRegion(frame);

    stateTime += delta;
    eatTime += delta;
  }

  /**
   * Render the frog to the screen.
   *
   * @param batch The SpriteBatch to render to
   */
  public void render(SpriteBatch batch, String location) {
    if (location.equals(Main.frogLocation)) {
      sprite.draw(batch);
    }
  }

  public Rectangle getBounds() {
    return new Rectangle(sprite.getX(), sprite.getY() + 2, 10, 4);
  }

  /** Eat player. */
  public void eat() {
    hasPlayer = true;
    escapeCounter = 0;
    eatTime = 0;
  }

  /**
   * Fling player out of frog.
   *
   * @param player the player character
   * @param delta time in seconds since last call
   */
  public void escape(Player player, float delta) {

    if (!escaping) {
      player.setPosition(sprite.getX(), sprite.getY() + sprite.getWidth() / 2);
      escaping = true;
      eatTime = 0;
    }
    player.move(-0.8f * delta * 4f, 0.1f * delta * 4f); // multiplier controls speed of slip

    player.setRotation(player.getRotation() + (360 * delta));
    if (eatTime > 1f) {
      hasPlayer = false;
      resetEscapeCounter();
      escaping = false;
      player.setRotation(0);
    }
  }

  /**
   * Return if the player is inside the frog.
   *
   * @return if player is in frog
   */
  public boolean hasPlayer() {
    return hasPlayer;
  }

  /** Increment number of escape attempts. */
  public void incrementEscapeCounter() {
    escapeCounter++;
  }

  /** Reset escape counter to zero. */
  public void resetEscapeCounter() {
    escapeCounter = 0;
  }

  /** Reset time since eaten. */
  public void resetEatCount() {
    eatTime = 0;
  }

  public int getEscapeCounter() {
    return escapeCounter;
  }

  /** Set timer for punch animation. */
  public void punch() {
    punched = 1.3f;
  }

  public boolean isEscaping() {
    return escaping;
  }
}
