package com.screens;

import static com.badlogic.mazegame.Main.WORLD_HEIGHT;
import static com.badlogic.mazegame.Main.WORLD_WIDTH;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.mazegame.Main;

/**
 * The game clock. Keeps track of how long is left until the end of the game.
 *
 * @author Wasif Mustafa
 */
public final class Timer {

  // Changed: store path in constant for testing
  public static final String IMG = "screens/timeOut.png";
  private static final String labelFormatString = "Time %02d:%02d";
  final Label label;
  private final Texture timeOut = new Texture(IMG);

  // Game Over time out Screen
  /** The amount of time remaining. */
  private float timeLeft;

  /** Whether the timer is currently ticking down. */
  private boolean isRunning = true;

  /** Construct a new timer for the game. */
  public Timer(float length) {
    this.timeLeft = length;
    label = new Label(getLabelString(), Main.labelStyle);
    label.setPosition(0.5f, -5);
    label.setFontScale(0.03f);
  }

  /** Constructs a mm:ss duration string to display on the label. */
  private String getLabelString() {
    if (timeLeft <= 0) {
      return "GAME OVER!";
    }
    int totalSeconds = Math.round((long) Math.floor(timeLeft));
    int minutes = Math.floorDiv(totalSeconds, 60);
    int seconds = Math.floorMod(totalSeconds, 60);
    return String.format(labelFormatString, minutes, seconds);
  }

  /** Update the timer based on the time since it was last updated. */
  public void tick() {
    float delta = Gdx.graphics.getDeltaTime();
    if (!isRunning) {
      return;
    }
    setTimeLeft(timeLeft - delta);
    if (timeLeft <= 0) {
      setTimeLeft(0);
      isRunning = false;
    }
  }

  /**
   * Render the timer to the screen.
   *
   * @param batch The SpriteBatch to render to.
   */
  public void render(SpriteBatch batch) {
    label.draw(batch, 255);
    if (timeLeft <= 0) {
      // Draw TimeOut Screen
      batch.draw(timeOut, 0, 0, WORLD_WIDTH, WORLD_HEIGHT);
      isRunning = false;
    }
  }

  /**
   * Checks whether the timer is currently running.
   *
   * @return A boolean value representing whether the timer is running
   */
  public boolean getIsRunning() {
    return isRunning;
  }

  /** Pauses the timer if currently running. */
  public void pause() {
    isRunning = false;
  }

  /** Resumes the timer if not currently running. */
  public void resume() {
    isRunning = true;
  }

  /**
   * Get time left of game.
   *
   * @return The time left on the timer
   */
  public double getTimeLeft() {
    return timeLeft;
  }

  /** Sets the value of the timeLeft property. */
  private void setTimeLeft(float timeLeft) {
    this.timeLeft = timeLeft;
    label.setText(getLabelString());
  }
}
