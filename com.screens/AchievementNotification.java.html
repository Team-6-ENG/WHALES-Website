package com.screens;

import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.GlyphLayout;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.mazegame.Main;
import com.rooms.WorldMap;

/**
 * Manages achievement unlock notifications displayed on screen.
 * [03/01/2026] - Added achievement notification display system
 * Uses BitmapFont for direct rendering to ensure proper world coordinate system */
@SuppressWarnings({"checkstyle:SummaryJavadoc", "checkstyle:JavadocContentLocation"})
public class AchievementNotification {
  private static final float DISPLAY_DURATION = 5f; // Show notification for 3 seconds
  private static final float FADE_IN_DURATION = 0.5f; // Fade in over 0.5 seconds
  private static final float FADE_OUT_DURATION = 0.5f; // Fade out over 0.5 seconds
  private String notificationText;
  private float displayTime = 0f;
  private boolean isActive = false;
  @SuppressWarnings("checkstyle:MemberName")
  private float xPos;
  @SuppressWarnings("checkstyle:MemberName")
  private float yPos;

  /**
   * Shows an achievement unlock notification
   *
   * @param achievement The achievement that was unlocked
   */
  public void show(Achievement achievement) {
    this.displayTime = 0f;
    this.isActive = true;

    // Create notification text: "You unlocked xxx - xxxxx"
    this.notificationText = "You unlocked " + achievement.name + "\n - " + achievement.description;

    // Position at top-left of screen - same as other labels for consistency
    this.xPos = WorldMap.BOUNDARY_WIDTH + 2;
    this.yPos = Main.WORLD_HEIGHT - 2f; // Very top, clearly visible
  }

  /**
   * Updates the notification (should be called every frame)
   *
   * @param delta Time since last frame
   */
  public void update(float delta) {
    if (isActive) {
      displayTime += delta;
      if (displayTime >= DISPLAY_DURATION) {
        isActive = false;
      }
    }
  }

  /**
   * Calculates the alpha (transparency) value based on fade in/out timing
   *
   * @return Alpha value between 0 (transparent) and 1 (opaque)
   */
  private float calculateAlpha() {
    if (displayTime < FADE_IN_DURATION) {
      // Fade in: 0 to 1
      return displayTime / FADE_IN_DURATION;
    } else if (displayTime > DISPLAY_DURATION - FADE_OUT_DURATION) {
      // Fade out: 1 to 0
      float fadeOutStart = DISPLAY_DURATION - FADE_OUT_DURATION;
      float fadeOutProgress = (displayTime - fadeOutStart) / FADE_OUT_DURATION;
      return 1f - fadeOutProgress;
    } else {
      // Fully visible
      return 1f;
    }
  }

  /**
   * Renders the notification if it's active
   *
   * @param batch The SpriteBatch to render to
   */
  public void render(SpriteBatch batch) {
    if (!isActive || notificationText == null || batch == null) {
      return;
    }

    // Calculate and apply alpha for fade effect
    float alpha = calculateAlpha();
    if (alpha <= 0.01f) {
      return;
    }

    // Get the font and set up for drawing
    BitmapFont font = Main.bitmapFont;
    if (font == null) {
      return;
    }

    // Set font scale and color with alpha
    // Font scale in world units
    float fontScale = 0.04f;
    font.getData().setScale(fontScale);
    font.setColor(Color.BLUE);

    // Calculate text width to ensure it fits on screen
    GlyphLayout layout = new GlyphLayout(font, notificationText);
    float textWidth = layout.width;
    float maxWidth = Main.WORLD_WIDTH - 6f; // Leave 3f margin on each side

    // If text is too wide, reduce font scale
    float actualFontScale;
    if (textWidth > maxWidth) {
      actualFontScale = fontScale * (maxWidth / textWidth);
      font.getData().setScale(actualFontScale);
      layout.setText(font, notificationText);
    }

    // Draw the text at the specified position
    font.draw(batch, notificationText, xPos, yPos);

    // Reset font scale to default for other uses
    font.getData().setScale(1f);
  }

  /**
   * Checks if a notification is currently being displayed
   *
   * @return true if notification is active, false otherwise
   */
  public boolean isActive() {
    return isActive;
  }
}
