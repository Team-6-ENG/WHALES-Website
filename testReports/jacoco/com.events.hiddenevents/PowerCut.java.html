package com.events.hiddenevents;

import com.badlogic.gdx.math.Vector2;

/** A hidden event which darkens the player's screen for a set length of time. */
public class PowerCut {
  /** The length of the power cut when triggered. */
  private final float duration;

  /** How dark the power cut currently is, 1 being pitch black, 0 being normal. */
  private float darknessLevel;

  /** How long has passed since the event was activated. */
  private float timer;

  private boolean triggered;
  private boolean active;

  /**
   * Construct a new hidden power cut event.
   *
   * @param position The position to trigger the event
   * @param duration The length of the power cut when triggered
   */
  public PowerCut(Vector2 position, float duration) {

    this.duration = duration;
    this.darknessLevel = 0f;
    this.timer = 0f;
  }

  /** Trigger event. */
  public void trigger() {
    // can only trigger event once
    if (triggered) {
      return;
    }

    triggered = true;
    darknessLevel = 1f;

    System.out.println("PowerCut trigger"); // for testing
  }

  /**
   * Update the event, ticking down timer and deactivating once event is over.
   *
   * @param delta The time since the last update
   */
  public void update(float delta) {
    if (!triggered) {
      return;
    }
    timer += delta;
    if (timer >= duration) {
      darknessLevel -= delta; // fades light back in
      if (darknessLevel <= 0f) {
        darknessLevel = 0f;
        reset();
        active = false;
      }
    }
  }

  /**
   * Get darkness level of room.
   *
   * @return <code>darknessLevel</code>
   */
  public float getDarknessLevel() {
    return darknessLevel;
  }

  /** Reset once power returns. */
  public void reset() {
    triggered = false;
    active = true;
    darknessLevel = 0f;
    timer = 0f;
  }

  public boolean isTriggered() {
    return triggered;
  }
}
