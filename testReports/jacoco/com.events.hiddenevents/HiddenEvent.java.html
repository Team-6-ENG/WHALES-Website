package com.events.hiddenevents;

import com.badlogic.gdx.math.Vector2;
import com.events.Event;

/**
 * Implementation of Event class for specific "hidden" events which are triggered without needing to
 * render anything.
 */
public abstract class HiddenEvent implements Event {
  /** No. of hidden events triggered. */
  private static int counter = 0;

  /** Position to activate the event. */
  protected Vector2 position;

  /** Whether the event has been triggered yet. */
  protected boolean triggered;

  /** Controls whether the event can be activated or not. */
  protected boolean active;

  /** Create a new hidden event at the given position. */
  protected HiddenEvent(Vector2 position) {
    this.position = position;
    this.triggered = false;
    this.active = true;
  }

  /**
   * Get number of hidden events triggered.
   *
   * @return <code>counter</code>
   */
  public static int getCounter() {
    return counter;
  }

  /** Add one to event counter. */
  public static void incrementCounter() {
    counter++;
  }

  /** Call when restarting game, resets event counter to zero. */
  public static void reset() {
    counter = 0;
  }

  @Override
  public Vector2 getPosition() {
    return position;
  }

  @Override
  public void setPosition(Vector2 position) {
    this.position = position;
  }

  @Override
  public boolean isTriggered() {
    return triggered;
  }

  @Override
  public void trigger() {
    if (!triggered) {
      this.triggered = true;
      counter++;
    }
  }

  /**
   * Is event active.
   *
   * @return <code>active</code>
   */
  public boolean isActive() {
    return active;
  }

  /** Deactivate the event so it can no longer be triggered. */
  public void deactivate() {
    this.active = false;
  }
}
