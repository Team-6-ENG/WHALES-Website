package com.events;

import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.mazegame.Main;
import com.rooms.WorldMap;
import java.util.ArrayList;
import java.util.Random;

/** Represents all parts of the duck statue. */
public class DuckStatue {

  public static final String HAT_IMG = "items/DuckTopHat.png";
  public static final String HEAD_IMG = "items/DuckHead.png";
  public static final String FEET_IMG = "items/DuckFeet.png";
  public static final String TORSO_IMG = "items/DuckTorso.png";
  public static final Integer NUM_AL_ROOMS = 7;
  private final DuckStatuePiece hat = new DuckStatuePiece(15, 3, 3f, 2.5f, "Nisa", HAT_IMG);
  private final DuckStatuePiece head = new DuckStatuePiece(15, 3, 3f, 2.5f, "Lake3", HEAD_IMG);
  private final DuckStatuePiece feet =
      new DuckStatuePiece(
          12, 3, 3f, 2.5f, "ALRooms" + ((new Random()).nextInt(NUM_AL_ROOMS) + 1), FEET_IMG);
  private final DuckStatuePiece torso =
      new DuckStatuePiece(12, 5, 3f, 2.5f, "Glasshouse", TORSO_IMG);

  /** Initialise duck statue manager. */
  public DuckStatue() {}

  /**
   * Check if every piece of the duck has been found.
   *
   * @return true if all found else false
   */
  public boolean checkAllFound() {
    return head.isTriggered() && feet.isTriggered() && torso.isTriggered();
  }

  /**
   * List all pieces of the duck in the game.
   *
   * @return ArrayList representing all duck pieces in the game
   */
  public ArrayList<DuckStatuePiece> getPieces() {
    ArrayList<DuckStatuePiece> pieces = new ArrayList<>();
    pieces.add(hat);
    pieces.add(head);
    pieces.add(feet);
    pieces.add(torso);
    return pieces;
  }

  /**
   * List pieces of duck which have been found.
   *
   * @return found pieces of duck
   */
  public ArrayList<DuckStatuePiece> getFoundPieces() {
    ArrayList<DuckStatuePiece> foundPieces = new ArrayList<>();
    if (head.isTriggered()) {
      foundPieces.add(head);
    }
    if (feet.isTriggered()) {
      foundPieces.add(feet);
    }
    if (torso.isTriggered()) {
      foundPieces.add(torso);
    }
    return foundPieces;
  }

  /**
   * Draw duck pieces to screen.
   *
   * @param batch current instance of sprite batch
   * @param currentRoom current player location
   * @param guardSleeping is guard sleeping
   */
  public void render(SpriteBatch batch, String currentRoom, boolean guardSleeping) {

    float jigsawY = ((float) Main.WORLD_HEIGHT / 2) + 3;
    float jigsawX = ((float) (Main.WORLD_WIDTH - (WorldMap.BOUNDARY_WIDTH / 2)) - 2f);
    head.render(batch, currentRoom, jigsawX, jigsawY, guardSleeping);
    torso.render(batch, currentRoom, jigsawX, jigsawY - 2.5f, guardSleeping);
    feet.render(batch, currentRoom, jigsawX, jigsawY - 5f, guardSleeping);

    // duck hat should be hidden until triggered
    if (currentRoom.equals(hat.getRoom()) && !hat.isTriggered()) {
      batch.draw(
          hat.getDuckTexture(),
          hat.getBounds().x,
          hat.getBounds().y,
          hat.getBounds().width,
          hat.getBounds().height);
    } else if (hat.isTriggered()) {
      batch.draw(hat.getDuckTexture(), jigsawX, jigsawY + 1.4f, 3, 2.5f);
    }
  }
}
