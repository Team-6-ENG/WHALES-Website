package com.events;

import com.badlogic.gdx.math.Vector2;

/** Event Interface for a triggered event in the world. */
public interface Event {

  /** Add one to negative event counter. */
  static void incrementNegativeEventCounter() {
    CounterHolder.negativeEventCount++;
  }

  /**
   * Get number of negative events found.
   *
   * @return number of negative events found
   */
  static int getNegativeEventCounter() {
    return CounterHolder.negativeEventCount;
  }

  /** Add one to positive events counter. */
  static void incrementPositiveEventCounter() {
    CounterHolder.positiveEventCounter++;
  }

  /**
   * Get number of positive events triggered.
   *
   * @return number of positive events found
   */
  static int getPositiveEventCounter() {
    return CounterHolder.positiveEventCounter;
  }

  /** Call when resetting the game, sets counters to zero. */
  static void reset() {
    CounterHolder.negativeEventCount = 0;
    CounterHolder.positiveEventCounter = 0;
  }

  /**
   * Get event position.
   *
   * @return The position of the event
   */
  Vector2 getPosition();

  /**
   * Sets the position of the event.
   *
   * @param position The new position
   */
  void setPosition(Vector2 position);

  /**
   * Check if event has been triggered.
   *
   * @return Whether the event has been triggered yet
   */
  boolean isTriggered();

  /** Trigger the event. */
  void trigger();

  /** An inner class to hold counter values for. */
  class CounterHolder {
    static int negativeEventCount = 0;
    static int positiveEventCounter = 0;
  }
}
