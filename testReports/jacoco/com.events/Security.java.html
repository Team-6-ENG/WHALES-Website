package com.events;

import com.badlogic.gdx.graphics.g2d.Sprite;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import com.badlogic.gdx.math.Rectangle;
import com.badlogic.mazegame.Main;
import com.characters.Animator;
import com.rooms.WorldMap;
import java.util.HashMap;

/** Represent campus security. */
public class Security extends Animator {

  static final String SHEET = "characters/security.png";
  Sprite sprite;
  private float speed = 8;
  private float stateTime;
  private boolean hasDuck;
  private boolean sleep;

  /** Security constructor. */
  public Security() {

    super(SHEET, 7, 2);

    HashMap<String, Integer[]> animationInfo = new HashMap<String, Integer[]>();
    // HashMap<String, Integer[]> animationInfo:
    //      key - Name of animation
    //      Value - Array representing row of animation on sprite sheet and index of start and end
    // frames
    animationInfo.put("walkRight", new Integer[] {0, 0, 7});
    animationInfo.put("sleep", new Integer[] {1, 0, 3});

    generateAnimation(animationInfo, 0.3f);

    sprite = new Sprite(animations.get("walkRight").getKeyFrame(0, true));
    sprite.setPosition(20, 10);
  }

  /** Update position and animation. */
  public void update(float delta) {
    if (!sleep) {
      if (sprite.getX() > Main.WORLD_WIDTH - WorldMap.BOUNDARY_WIDTH - (sprite.getWidth() * 2)) {
        speed *= -1;
        sprite.setX(Main.WORLD_WIDTH - WorldMap.BOUNDARY_WIDTH - 0.2f - (sprite.getWidth() * 2));
      } else if ((sprite.getX() < WorldMap.BOUNDARY_WIDTH + sprite.getWidth())) {
        speed *= -1;
        sprite.setX(WorldMap.BOUNDARY_WIDTH + sprite.getWidth() + 1.2f);
      }

      TextureRegion frame = animations.get("walkRight").getKeyFrame(stateTime, true);

      if ((speed < 0 && !frame.isFlipX()) || (speed > 0 && frame.isFlipX())) {
        frame.flip(true, false);
      }

      sprite.setSize(3f, 4f);
      sprite.setRegion(frame);
      sprite.setX(sprite.getX() + (speed * delta));
    } else {
      TextureRegion frame = animations.get("sleep").getKeyFrame(stateTime, true);

      sprite.setRegion(frame);

      sprite.setPosition(10, 5);
      sprite.setSize(3f, 4f);
    }

    stateTime += delta;
  }

  /**
   * Render the guard to the screen.
   *
   * @param batch The SpriteBatch to render to
   */
  public void render(SpriteBatch batch, String location) {
    if (location.equals(Main.securityLocation)) {
      sprite.draw(batch);
    }
  }

  public void setHasDuck(boolean hasDuck) {
    this.hasDuck = hasDuck;
  }

  /**
   * Has security stolen duck.
   *
   * @return if guard has stolen duck
   */
  public boolean hasDuck() {
    return hasDuck;
  }

  /**
   * Call when changing rooms to make guard sleep if duck stolen.
   *
   * @param location current location of player
   */
  public void onRoomChange(String location) {
    if (location.equals(Main.securityLocation) && hasDuck) {
      sleep = true;
    }
  }

  public boolean isSleeping() {
    return sleep;
  }

  public Rectangle getBounds() {
    return sprite.getBoundingRectangle();
  }
}
