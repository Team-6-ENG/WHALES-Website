package com.events;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector2;
import com.characters.Player;

/**
 * Slip on coffee negative event .
 *
 * @author Wasif Mustafa A coffee causes the player to slip when stepped on
 */
public class SlipOnCoffee extends Obstacle {
  // Changed: store path in constant for testing
  public static final String IMG = "items/Coffee.png";
  public final float duration = 1f; // length of slip
  private final Texture coffeeTexture;
  private boolean triggered = false;
  private boolean active = false;
  private float timer = 0;
  private Vector2 slipDirection;

  /** Constructor. */
  public SlipOnCoffee(float x, float y, float width, float height) {
    super(x, y, width, height);
    coffeeTexture = new Texture(Gdx.files.internal(IMG));
  }

  /** Called each frame to update the slip state. */
  public void update(float delta, Player player, String currentRoom, String targetRoom) {

    timer += delta;
    if (!currentRoom.equals(targetRoom)) {
      reset();
      return;
    }
    if (!active) {
      return;
    }

    if (timer < duration) {
      player.move(
          slipDirection.x * delta * 4f,
          slipDirection.y * delta * 4f); // multiplier controls speed of slip

      player.setRotation(player.getRotation() + (360 * delta));
    } else if (timer >= duration * 2) {

      // Slip finished
      reset();
      player.setRotation(0f);
    }
  }

  /** Call to trigger event. */
  public void trigger() {

    if (!triggered) {
      triggered = true;
      Event.incrementNegativeEventCounter();
    }
    if (!active) {
      active = true;
      timer = 0;
      slipDirection = new Vector2((float) Math.random() - 0.5f, (float) Math.random() - 0.5f).nor();
    }
  }

  /** Render coffee. */
  @Override
  public void render(SpriteBatch batch) {
    batch.draw(coffeeTexture, getBounds().x, getBounds().y, getBounds().width, getBounds().height);
  }

  /** Reset once room is left. */
  public void reset() {
    active = false;
  }

  /**
   * If event is active.
   *
   * @return returns if event has been triggered
   */
  public boolean isActive() {
    return active;
  }
}
