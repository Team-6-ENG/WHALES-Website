package com.events;

import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.mazegame.Main;
import com.events.hiddenevents.HiddenEvent;

/**
 * Represents each piece of duck statue.
 *
 * @author Wasif Mustafa A Duck Stautue resembling longboi is interacted with causing the player to
 *     win Changed 21/12/2025 from DuckStatue to DuckStatuePiece
 */
public class DuckStatuePiece extends Obstacle {

  private final Texture duckStatueTexture;
  private final String room;
  private boolean triggered = false;
  private boolean stolen = false;

  /** Sets position of the obstacle and size loads texture of the statue. */
  public DuckStatuePiece(float x, float y, float width, float height, String room, String texture) {
    super(x, y, width, height);

    duckStatueTexture = new Texture(texture);
    this.room = room;
  }

  /**
   * Has piece been collected.
   *
   * @return returns if event has been triggered
   */
  public boolean isTriggered() {
    return triggered;
  }

  /** Triggers event. */
  public void trigger() {

    triggered = true;
    if (duckStatueTexture.toString().equals(DuckStatue.HAT_IMG)) {
      HiddenEvent.incrementCounter();
    }
  }

  /** Used when campus security steals your duck pieces. */
  public void steal() {
    stolen = true;
  }

  public boolean isStolen() {
    return stolen;
  }

  /** Take piece back from campus security. */
  public void unsteal() {
    stolen = false;
  }

  /**
   * Check is piece in same room as player.
   *
   * @param currentRoom location of player
   * @return if in same room
   */
  public boolean inRoom(String currentRoom) {
    return room.equals(currentRoom);
  }

  public String getRoom() {
    return room;
  }

  public Texture getDuckTexture() {
    return duckStatueTexture;
  }

  /** Render duck. */
  public void render(
      SpriteBatch batch, String currentRoom, float jigsawX, float jigsawY, boolean guardSleeping) {
    if (inRoom(currentRoom) && !triggered) {
      batch.draw(
          duckStatueTexture, getBounds().x, getBounds().y, getBounds().width, getBounds().height);
    } else if (stolen && currentRoom.equals(Main.securityLocation) && guardSleeping) {
      batch.draw(
          duckStatueTexture, getBounds().x, getBounds().y, getBounds().width, getBounds().height);
    }

    if (triggered && !stolen) {
      batch.draw(duckStatueTexture, jigsawX, jigsawY, 3, 2.5f);
    } else {
      batch.setColor(0, 0, 0, 0.5f);
      batch.draw(duckStatueTexture, jigsawX, jigsawY, 3, 2.5f);
      batch.setColor(1, 1, 1, 1);
    }
  }
}
