package com.events;

/**
 * An Item that can be interacted with to cause some effect. Either a consumable which runs it's
 * effect as soon as it's triggered, or an item that can be picked up and placed into the player's
 * inventory (not yet implemented). <br>
 * <br>
 * After creating an item, it must be placed in the <code>items</code> <code>ArrayList</code> of
 * <code>WorldScreen</code> to handle the logic.
 */
public class Item extends Obstacle {
  private final ItemType type;
  private final Runnable useEffect;

  /**
   * Create a new item.
   *
   * @param x The x position of the obstacle.
   * @param y The y position of the obstacle.
   * @param textureInternalPath The path to the texture to render for this obstacle
   * @param useEffect Function to call when the item is used
   */
  public Item(
      float x,
      float y,
      float width,
      float height,
      String textureInternalPath,
      ItemType type,
      Runnable useEffect) {
    super(x, y, width, height, textureInternalPath);
    super.setSolid(false);

    this.type = type;
    this.useEffect = useEffect;
  }

  @Override
  public void trigger() {
    super.trigger();
    useEffect.run();
    dispose();
  }
}
