package com.screens;

import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.mazegame.Main;

/** Tracks player score. */
public final class Score {

  // Bonus for finding the duck
  private static final int DUCK_BONUS = 150;
  private static final String LABEL_FORMAT = "Score: %d";
  // Label to display the score
  final Label label;
  private int currentScore = 0;

  /** Score constructor. */
  public Score() {
    label = new Label(String.format(LABEL_FORMAT, currentScore), Main.labelStyle);
    label.setPosition(Main.WORLD_WIDTH - 6, Main.WORLD_HEIGHT - 8);
    label.setFontScale(0.03f);
  }

  private void updateLabel() {
    label.setText(String.format(LABEL_FORMAT, currentScore));
  }

  /**
   * Add points to score.
   *
   * @param points number of points to add
   */
  public void addPoints(int points) {
    currentScore += points;
    updateLabel();
  }

  /**
   * Take points awau from score.
   *
   * @param points number of points to subtract
   */
  public void subtractPoints(int points) {
    currentScore = Math.max(0, currentScore - points);
    updateLabel();
  }

  /** Reset score to zero. */
  public void reset() {
    currentScore = 0;
    updateLabel();
  }

  public int getCurrentScore() {
    return currentScore;
  }

  /** Called when player finds the duck statue, adds bonus to score. */
  public void addDuckBonus() {
    addPoints(DUCK_BONUS);
  }

  /** Called at the very end (win or lose) to add remaining seconds as points. */
  public void addTimeRemainingBonus(double timeLeftInSeconds) {
    int timeBonus = (int) timeLeftInSeconds; // direct 1:1 seconds to points
    addPoints(timeBonus);
  }

  /**
   * Draw score on screen.
   *
   * @param batch current sprite batch
   */
  public void render(SpriteBatch batch) {
    label.draw(batch, 1f);
  }
}
