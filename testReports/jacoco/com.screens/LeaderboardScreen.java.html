package com.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.Screen;
import com.badlogic.gdx.ScreenAdapter;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.ChangeListener;
import com.badlogic.gdx.utils.viewport.ScreenViewport;
import com.badlogic.mazegame.Main;
import java.util.ArrayList;
import java.util.List;

/** [21/12/2025] - Added this screen class to display the leaderboard scores. */
public class LeaderboardScreen extends ScreenAdapter {
  private static final float BACK_BUTTON_ORIGINAL_WIDTH = 200f;
  private static final float BACK_BUTTON_ORIGINAL_HEIGHT = 60f;
  private final Main game;
  private final LeaderboardManager leaderboardManager;
  private final Screen returnScreen;
  private List<Label> scoreLabels;
  private Label titleLabel;
  private Stage stage;
  private TextButton backButton;

  /**
   * Initialise leaderboard screen.
   *
   * @param game current instance of game
   * @param returnScreen screen to return to
   */
  public LeaderboardScreen(Main game, Screen returnScreen) {
    this.game = game;
    this.leaderboardManager = new LeaderboardManager();
    this.returnScreen = returnScreen;
  }

  @Override
  public void show() {
    stage = new Stage(new ScreenViewport());
    Gdx.input.setInputProcessor(stage);

    if (scoreLabels != null) {
      scoreLabels.clear();
    }

    titleLabel = new Label("LEADERBOARD", Main.labelStyle);
    titleLabel.setColor(new Color(0.85f, 0.85f, 0.85f, 1f));

    scoreLabels = new ArrayList<>();

    List<ScoreEntry> scores = leaderboardManager.getTopScores(5);
    if (scores.isEmpty()) {
      Label noScoresLabel =
          new Label("No scores yet. Play the game to set a record!", Main.labelStyle);
      noScoresLabel.setColor(new Color(0.9f, 0.9f, 0.9f, 1f));
      scoreLabels.add(noScoresLabel);
    } else {
      Color[] rankColors = {
        new Color(0.6f, 1f, 0.7f, 1f), // Light green
        new Color(1f, 0.85f, 0.6f, 1f), // Light orange
        new Color(0.9f, 0.9f, 0.9f, 1f), // Light gray
        new Color(0.7f, 0.9f, 1f, 1f), // Light blue
        new Color(0.95f, 0.8f, 1f, 1f) // Light purple
      };

      for (int i = 0; i < scores.size(); i++) {
        ScoreEntry entry = scores.get(i);
        int rank = i + 1;

        String labelText =
            String.format(
                "No.%d  %-15s  Score: %-6d  Time:  %s",
                rank, entry.playerName, entry.score, entry.date != null ? entry.date : "");

        Label scoreLabel = new Label(labelText, Main.labelStyle);
        Color labelColor =
            i < rankColors.length ? rankColors[i] : rankColors[rankColors.length - 1];
        scoreLabel.setColor(labelColor);
        scoreLabels.add(scoreLabel);
      }
    }

    backButton = new TextButton("Back", Main.uiSkin);
    backButton.setSize(BACK_BUTTON_ORIGINAL_WIDTH, BACK_BUTTON_ORIGINAL_HEIGHT);
    if (backButton.getLabel() != null) {
      backButton.getLabel().setWrap(true);
    }
    backButton.addListener(
        new ChangeListener() {
          @Override
          public void changed(ChangeEvent event, com.badlogic.gdx.scenes.scene2d.Actor actor) {
            game.setScreen(returnScreen);
          }
        });

    stage.addActor(titleLabel);
    for (Label label : scoreLabels) {
      stage.addActor(label);
    }
    stage.addActor(backButton);

    updatePositions(Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
  }

  private void updatePositions(float screenWidth, float screenHeight) {
    float centerX = screenWidth / 2f;

    float scaleX =
        screenWidth > 0 && Main.VIEWPORT_SIZE.x > 0 ? screenWidth / Main.VIEWPORT_SIZE.x : 1f;
    float scaleY =
        screenHeight > 0 && Main.VIEWPORT_SIZE.y > 0 ? screenHeight / Main.VIEWPORT_SIZE.y : 1f;
    float uniformScale = Math.max(0.1f, Math.min(scaleX, scaleY));

    if (titleLabel != null) {
      titleLabel.setFontScale(uniformScale);
      titleLabel.pack();
      titleLabel.setPosition(centerX - titleLabel.getWidth() / 2f, screenHeight * 0.88f);
    }

    float startY = screenHeight * 0.72f;
    float lineSpacing = screenHeight * 0.11f;

    for (int i = 0; i < scoreLabels.size(); i++) {
      Label label = scoreLabels.get(i);
      if (label != null) {
        label.setFontScale(0.6f * uniformScale);
        label.pack();
        float y = startY - (i * lineSpacing);
        label.setPosition(centerX - label.getWidth() / 2f, y);
      }
    }

    if (backButton != null) {
      backButton.setSize(
          BACK_BUTTON_ORIGINAL_WIDTH * uniformScale, BACK_BUTTON_ORIGINAL_HEIGHT * uniformScale);
      if (backButton.getLabel() != null) {
        float buttonHeight = BACK_BUTTON_ORIGINAL_HEIGHT * uniformScale;
        float buttonWidth = BACK_BUTTON_ORIGINAL_WIDTH * uniformScale;
        float maxFontScaleByHeight = (buttonHeight * 0.6f) / 15f;
        float maxFontScaleByWidth = (buttonWidth * 0.8f) / 40f;
        float buttonFontScale =
            Math.min(uniformScale * 1.5f, Math.min(maxFontScaleByHeight, maxFontScaleByWidth));
        backButton.getLabel().setFontScale(buttonFontScale);
        backButton.getLabel().setWrap(true);
      }
      backButton.setPosition(centerX - backButton.getWidth() / 2f, screenHeight * 0.08f);
    }
  }

  @Override
  public void render(float delta) {
    Gdx.gl.glClearColor(0.15f, 0.18f, 0.25f, 1);
    Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);

    game.getViewport().apply();
    stage.act(delta);

    com.badlogic.gdx.graphics.g2d.SpriteBatch batch = game.getBatch();
    batch.setProjectionMatrix(stage.getCamera().combined);

    batch.begin();
    if (titleLabel != null && titleLabel.isVisible()) {
      float screenWidth = Gdx.graphics.getWidth();
      float screenHeight = Gdx.graphics.getHeight();
      float centerX = screenWidth / 2f;
      float titleY = screenHeight * 0.88f;
      float outlineOffset = 3f;

      titleLabel.setColor(new Color(0f, 0f, 0f, 0.8f));
      for (int dx = -1; dx <= 1; dx++) {
        for (int dy = -1; dy <= 1; dy++) {
          if (dx != 0 || dy != 0) {
            titleLabel.setPosition(
                centerX - titleLabel.getWidth() / 2f + dx * outlineOffset,
                titleY + dy * outlineOffset);
            titleLabel.draw(batch, 1f);
          }
        }
      }

      titleLabel.setColor(new Color(0.85f, 0.85f, 0.85f, 1f));
      titleLabel.setPosition(centerX - titleLabel.getWidth() / 2f, titleY);
      titleLabel.draw(batch, 1f);
    }
    batch.end();

    boolean titleWasVisible = titleLabel != null && titleLabel.isVisible();
    if (titleLabel != null) {
      titleLabel.setVisible(false);
    }
    stage.draw();
    if (titleLabel != null) {
      titleLabel.setVisible(titleWasVisible);
    }
    if (Gdx.input.isKeyJustPressed(Input.Keys.ESCAPE)) {
      game.setScreen(new HomeScreen(game));
    }
  }

  @Override
  public void resize(int width, int height) {
    if (stage != null) {
      stage.getViewport().update(width, height, true);
      updatePositions(width, height);
    }
  }

  @Override
  public void dispose() {
    if (scoreLabels != null) {
      scoreLabels.clear();
    }
    if (stage != null) {
      stage.dispose();
    }
  }
}
