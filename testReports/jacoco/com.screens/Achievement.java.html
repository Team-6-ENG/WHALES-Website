package com.screens;

/**
 * A data class representing a single achievement. Stores the achievement's identification,
 * description, and unlock status.
 * [01/01/2026] - Added achievements
 */
public class Achievement {
  /** The unique identifier for the achievement. */
  public String id;

  /** The display name of the achievement. */
  public String name;

  /** The description of what the achievement requires. */
  public String description;

  /** Whether the achievement has been unlocked. */
  public boolean unlocked;

  /** The date when the achievement was unlocked (formatted as a string, null if not unlocked). */
  public String unlockedDate;

  /** Default constructor required for JSON deserialization. */
  public Achievement() {}

  /**
   * Constructs a new Achievement with the given properties.
   *
   * @param id The unique identifier for the achievement
   * @param name The display name of the achievement
   * @param description The description of what the achievement requires
   */
  public Achievement(String id, String name, String description) {
    this.id = id;
    this.name = name;
    this.description = description;
    this.unlocked = false;
    this.unlockedDate = null;
  }

  /**
   * Unlocks this achievement and sets the unlock date.
   *
   * @param date The date when the achievement was unlocked
   */
  public void unlock(String date) {
    this.unlocked = true;
    this.unlockedDate = date;
  }

  @Override
  public String toString() {
    return String.format(
        "Achievement<id: %s, name: %s, unlocked: %s, date: %s>", id, name, unlocked, unlockedDate);
  }
}
