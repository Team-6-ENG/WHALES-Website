package com.screens;

import java.util.ArrayList;
import java.util.List;

/**
 * A wrapper class for storing leaderboard data in JSON format. This class is used for serialization
 * and deserialization of the leaderboard. [20/12/2025] - Added this class to store the list of
 * scores for JSON serialization
 */
public class LeaderboardData {
  /** The list of score entries. */
  public List<ScoreEntry> scores;

  /** Default constructor required for JSON deserialization. */
  public LeaderboardData() {
    this.scores = new ArrayList<>();
  }

  /**
   * Constructs a new LeaderboardData with the given list of scores.
   *
   * @param scores The list of score entries
   */
  public LeaderboardData(List<ScoreEntry> scores) {
    this.scores = scores != null ? scores : new ArrayList<>();
  }
}
