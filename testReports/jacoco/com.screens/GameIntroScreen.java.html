package com.screens;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.ScreenAdapter;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.GL20;
import com.badlogic.gdx.scenes.scene2d.Stage;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.ChangeListener;
import com.badlogic.gdx.utils.viewport.ScreenViewport;
import com.badlogic.mazegame.Main;
import com.rooms.WorldMap;

/** Game intro and explanation screen. [05/01/2026]-Added this screen for context */
public class GameIntroScreen extends ScreenAdapter {
  private static final float BUTTON_WIDTH = 180f;
  private static final float BUTTON_HEIGHT = 60f;
  private final Main game;
  private final String sheet;
  private final String img;
  private Stage stage;
  private Label titleLabel;
  private TextButton playButton;

  /**
   * Initialise screen.
   *
   * @param game current instance of Main
   * @param sheet selected sprite sheet
   * @param img selected sprite image
   */
  public GameIntroScreen(Main game, String sheet, String img) {
    this.game = game;
    this.sheet = sheet;
    this.img = img;
  }

  @Override
  public void show() {
    stage = new Stage(new ScreenViewport());
    Gdx.input.setInputProcessor(stage);

    titleLabel =
        new Label(
            "The duck statue has been stolen and hidden around campus. "
                + "Currently you are to blame and must reassemble the statue to prove your "
                + "innocence. Don't let campus security catch you with part of the statue "
                + "though as this may reinforce their mistrust.",
            Main.labelStyle);

    titleLabel.setColor(new Color(1f, 1f, 0f, 1f));
    titleLabel.setWrap(true);
    titleLabel.setWidth(Main.WORLD_WIDTH - (WorldMap.BOUNDARY_WIDTH * 2f));

    playButton = new TextButton("Play", Main.uiSkin);
    playButton.setSize(BUTTON_WIDTH, BUTTON_HEIGHT);
    if (playButton.getLabel() != null) {
      playButton.getLabel().setWrap(true);
    }
    playButton.addListener(
        new ChangeListener() {
          @Override
          public void changed(ChangeEvent event, com.badlogic.gdx.scenes.scene2d.Actor actor) {
            game.setScreen(new WorldScreen(game, sheet, img));
          }
        });

    stage.addActor(titleLabel);
    stage.addActor(playButton);

    updatePositions(Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
  }

  private void updatePositions(float screenWidth, float screenHeight) {
    float centerX = screenWidth / 2f;

    float scaleX =
        screenWidth > 0 && Main.VIEWPORT_SIZE.x > 0 ? screenWidth / Main.VIEWPORT_SIZE.x : 1f;
    float scaleY =
        screenHeight > 0 && Main.VIEWPORT_SIZE.y > 0 ? screenHeight / Main.VIEWPORT_SIZE.y : 1f;
    float fontScale = Math.max(0.1f, Math.min(scaleX, scaleY));

    float titleY = screenHeight * 0.6f;

    if (titleLabel != null) {
      titleLabel.setWidth(screenWidth * 0.9f);
      titleLabel.setFontScale(fontScale);
      titleLabel.setPosition(centerX - titleLabel.getWidth() / 2f, titleY);
    }

    if (playButton != null) {
      playButton.setSize(BUTTON_WIDTH * fontScale, BUTTON_HEIGHT * fontScale);
      if (playButton.getLabel() != null) {
        float buttonHeight = BUTTON_HEIGHT * fontScale;
        float buttonWidth = BUTTON_WIDTH * fontScale;
        float maxFontScaleByHeight = (buttonHeight * 0.6f) / 15f;
        float maxFontScaleByWidth = (buttonWidth * 0.8f) / 30f;
        float buttonFontScale =
            Math.min(fontScale * 1.5f, Math.min(maxFontScaleByHeight, maxFontScaleByWidth));
        playButton.getLabel().setFontScale(buttonFontScale);
        playButton.getLabel().setWrap(true);
      }
      playButton.setPosition(centerX - playButton.getWidth() / 2f, screenHeight * 0.08f);
    }
  }

  @Override
  public void render(float delta) {
    Gdx.gl.glClearColor(0.15f, 0.18f, 0.25f, 1);
    Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);

    game.getViewport().apply();
    stage.act(delta);

    stage.draw();

    if (Gdx.input.isKeyJustPressed(Input.Keys.ESCAPE)) {
      game.setScreen(new HomeScreen(game));
    }
  }

  @Override
  public void resize(int width, int height) {
    if (stage != null) {
      stage.getViewport().update(width, height, true);
      updatePositions(width, height);
    }
  }

  @Override
  public void dispose() {
    if (stage != null) {
      stage.dispose();
    }
  }
}
