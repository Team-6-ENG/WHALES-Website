package com.screens;

import java.util.ArrayList;
import java.util.List;

/**
 * A wrapper class for storing achievement data in JSON format. This class is used for serialization
 * and deserialization of achievements.
 * [01/01/2026] - Added class for JSON serialization
 */
public class AchievementData {
  /** The list of achievement entries. */
  public List<Achievement> achievements;

  /** Default constructor required for JSON deserialization. */
  public AchievementData() {
    this.achievements = new ArrayList<>();
  }

  /**
   * Constructs a new AchievementData with the given list of achievements.
   *
   * @param achievements The list of achievement entries
   */
  public AchievementData(List<Achievement> achievements) {
    this.achievements = achievements != null ? achievements : new ArrayList<>();
  }
}
