package com.screens;

/**
 * A data class representing a single entry in the leaderboard. Stores the time remaining when the
 * player completed the game and the date of completion. [21/12/2025] - Added store individual
 * leaderboard score entries (time remaining and date)
 */
public class ScoreEntry {
  /** The time remaining (in seconds) when the player completed the game. */
  public int score;

  /** The date when the score was achieved (formatted as a string). */
  public String date;

  public float timeLeft;
  // Change [04/01/2026] - Added player name to store player's name for leaderboard display
  public String playerName;

  /** Default constructor required for JSON deserialization. */
  public ScoreEntry() {}

  /**
   * Constructs a new ScoreEntry with the given time remaining and date.
   *
   * @param score The final score accumulated when the game is completed
   * @param date The date when the score was achieved
   * @param time The time remaining when the game was completed
   */
  public ScoreEntry(int score, String date, float time) {
    this.score = score;
    this.date = date;
    this.timeLeft = time;
    this.playerName = "Player"; // Default name for backward compatibility
  }

  /**
   * Constructs a new ScoreEntry with the given time remaining, date, and player name.
   *
   * @param score The final score accumulated when the game is completed
   * @param date The date when the score was achieved
   * @param time The time remaining when the game was completed
   * @param playerName The name of the player who achieved this score
   */
  public ScoreEntry(int score, String date, float time, String playerName) {
    this.score = score;
    this.date = date;
    this.timeLeft = time;
    this.playerName =
        playerName != null && !playerName.trim().isEmpty() ? playerName.trim() : "Player";
  }

  @Override
  public String toString() {
    return String.format(
        "ScoreEntry<playerName: %s, totalScore: %d, date: %s>", playerName, score, date);
  }
}
