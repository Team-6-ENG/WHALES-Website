package com.rooms;

/** An obstacle encoded in the map.json file, to be converted into an obstacle in world. */
public class JsonObstacle {
  public int obstacleX;
  public int obstacleY;
  public int width;
  public int height;
  public String textureInternalPath;

  @Override
  public String toString() {
    return String.format(
        "JsonObstacle<x: %d, y: %d, width: %d, height: %d, textureInternalPath: %s",
        obstacleX, obstacleY, width, height, textureInternalPath);
  }
}
