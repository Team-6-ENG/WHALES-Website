package com.rooms;

import java.util.Arrays;

/** A room encoded in the map.json file. */
public class JsonRoom {
  public String left;
  public String right;
  public String up;
  public String down;
  public JsonObstacle[] obstacles;

  /**
   * Return whether there is a room in given direction.
   *
   * @param direction direction to check
   * @return if room exists in direction
   */
  public String getNext(OrthogonalDirection direction) {
    switch (direction) {
      case left:
        return this.left;
      case right:
        return this.right;
      case up:
        return this.up;
      case down:
        return this.down;
      default:
        return null;
    }
  }

  @Override
  public String toString() {
    return String.format(
        "Room<Left: %s, Right: %s, Up: %s, Down: %s, Obstacles: %s",
        left, right, up, down, Arrays.toString(obstacles));
  }
}
