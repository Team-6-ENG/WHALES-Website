package com.badlogic.mazegame;

import com.badlogic.gdx.Game;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.math.Vector2;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.Skin;
import com.badlogic.gdx.utils.viewport.FitViewport;
import com.events.Event;
import com.events.hiddenevents.HiddenEvent;
import com.screens.HomeScreen;

/** Runs the game. */
public class Main extends Game {
  // TODO: Refactor WORLD_WIDTH and WORLD_HEIGHT to use a Vector2 WORLD_SIZE
  /** The width of the world box in arbitrary units. */
  public static final int WORLD_WIDTH = 40;

  /** The height of the world box in arbitrary units. */
  public static final int WORLD_HEIGHT = 25;

  /** The size of the viewport. */
  public static final Vector2 VIEWPORT_SIZE = new Vector2(800, 500);

  /** Default `BitmapFont` for text. */
  public static BitmapFont bitmapFont;

  /** Default `LabelStyle` for labels. */
  public static Label.LabelStyle labelStyle;

  // Change [17/12/2025] - Added shared Skin resource for UI buttons using existing uiskin
  public static Skin uiSkin;
  public static String coffeeSlipRoom = "RCH";
  public static String professorRoom = "MainPath3";
  public static String securityLocation = "LakePath9";
  public static String mapRoom = "Law";
  public static String frogLocation = "LakePath13";
  public String vendingMachineRoom = "DavidKato";
  private SpriteBatch spriteBatch;
  private FitViewport viewport;
  // Change [04/01/2026] - Added player name to store current player's name for leaderboard
  private String playerName = "";

  public String getPlayerName() {
    return playerName;
  }

  public void setPlayerName(String name) {
    this.playerName = name != null && !name.trim().isEmpty() ? name.trim() : "Player";
  }

  @Override
  public void create() {
    try {
      viewport = new FitViewport(WORLD_WIDTH, WORLD_HEIGHT);
      spriteBatch = new SpriteBatch();

      // Initialise font objects
      try {
        bitmapFont = new BitmapFont(Gdx.files.internal("font/font.fnt"));
        bitmapFont.getData().setScale(0.4f);
      } catch (Exception e) {
        Gdx.app.error("Main", "Failed to load font/font.fnt, creating default font", e);
        bitmapFont = new BitmapFont();
      }
      // Allows font spacing to be smooth in low world sizes
      bitmapFont.setUseIntegerPositions(false);
      labelStyle = new Label.LabelStyle(bitmapFont, new Color(255, 255, 255, 255));
      // Change [17/12/2025] - Load uiskin resource for buttons
      // Change [01/01/2026] - Added try-catch for uiskin loading because we need to handle resource
      // loading failures gracefully
      try {
        uiSkin = new Skin(Gdx.files.internal("buttons/uiskin.json"));
      } catch (Exception e) {
        Gdx.app.error("Main", "Failed to load uiskin.json, creating default skin", e);
        // Create a default skin if uiskin fails to load
        uiSkin = new Skin();
        uiSkin.add("default-font", bitmapFont);
      }
      // Game will start from the HomeScreen now
      setScreen(new HomeScreen(this));
    } catch (Exception e) {
      Gdx.app.error("Main", "Fatal error during initialization", e);
      throw e; // Re-throw to let the application handle it
    }
  }

  public SpriteBatch getBatch() {
    return spriteBatch;
  }

  public FitViewport getViewport() {
    return viewport;
  }

  /** Reset game variables. */
  public void resetGame() {
    Event.reset();
    HiddenEvent.reset();
  }

  @Override
  public void resize(int width, int height) {
    if (width > 0 && height > 0 && viewport != null) {
      viewport.update(width, height, true);
    }
    super.resize(width, height);
  }

  @Override
  public void dispose() {
    spriteBatch.dispose();
    // Change [17/12/2025] - Dispose uiSkin resource
    if (uiSkin != null) {
      uiSkin.dispose();
    }
    super.dispose();
  }
}
