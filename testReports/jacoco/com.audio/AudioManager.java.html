package com.audio;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.audio.Music;
import com.badlogic.gdx.audio.Sound;

/** Added 24.11.25 for testability and encapsulation. */
public class AudioManager {
  static final String MUSIC = "sounds/background-music.mp3";
  static final String COFFEE = "sounds/goofy-slip.mp3";
  static final String DRINKING = "sounds/drinking-sound.mp3";
  static final String COLLECT = "sounds/pick-up-sound.mp3";
  static final String WIND = "sounds/wind-sound.mp3";
  static final String POWER_CUT = "sounds/power-cut.mp3";
  static final String SWIPE_SOUND = "sounds/swipe.mp3";
  static final String FROG_SOUND = "sounds/frog.mp3";
  static final String GULP = "sounds/gulp.mp3";
  private final Music music;
  private final Sound coffee;
  private final Sound drinking;
  private final Sound collectItem;
  private final Sound wind;
  private final Sound powerCut;
  private final Sound swipe;
  private final Sound frog;
  private final Sound gulp;
  private long windId = 0;
  // Change [17/12/2025] - Store volume level (0.0 to 1.0) for volume control
  private float volume = 0.7f;
  private float musicVolume = 0.5f;
  private float preMuteVolume = volume;
  private float preMuteMusicVolume = musicVolume;

  /** Controls sounds and volume of the game. */
  public AudioManager() {
    music = Gdx.audio.newMusic(Gdx.files.internal(MUSIC));
    // Change [17/12/2025] - Set initial volume
    music.setVolume(musicVolume);
    coffee = Gdx.audio.newSound(Gdx.files.internal(COFFEE));
    drinking = Gdx.audio.newSound(Gdx.files.internal(DRINKING));
    collectItem = Gdx.audio.newSound(Gdx.files.internal(COLLECT));
    wind = Gdx.audio.newSound(Gdx.files.internal(WIND));
    powerCut = Gdx.audio.newSound(Gdx.files.internal(POWER_CUT));
    frog = Gdx.audio.newSound(Gdx.files.internal(FROG_SOUND));
    gulp = Gdx.audio.newSound(Gdx.files.internal(GULP));
    swipe = Gdx.audio.newSound(Gdx.files.internal(SWIPE_SOUND));
  }

  /** Start game music on loop. */
  public void playMusic() {
    music.play();
    music.setLooping(true);
  }

  /** Play coffee slip sound. */
  public void playCoffee() {
    coffee.play(volume);
  }

  /** Play drinking sound. */
  public void playDrinking() {
    drinking.play(volume);
  }

  /** Play item collection sound. */
  public void playCollectItem() {
    collectItem.play(volume);
  }

  /** Play frog croak sound. */
  public void playFrogSound() {
    frog.play(volume);
  }

  /** Play frog gulp sound. */
  public void playGulp() {
    gulp.play(volume * 4);
  }

  /** Start wind sound on loop. */
  public void playWind() {
    windId = wind.loop(volume);
  }

  /** Stop wind sound. */
  public void stopWind() {
    wind.stop();
    windId = 0;
  }

  /** Play power cut sound. */
  public void playPowerCut() {
    powerCut.play(volume);
  }

  /** Play swipe sound. */
  public void playSwipe() {
    swipe.play(volume);
  }

  /** Stop game music. */
  public void stopMusic() {
    if (music.isPlaying()) {
      music.stop();
    }
  }

  /** Mute and unmute all game sounds. */
  public void toggleMute() {
    if (volume != 0 || musicVolume != 0) {
      preMuteVolume = volume;
      preMuteMusicVolume = musicVolume;
      setMusicVolume(0);
      setVolume(0);
    } else {
      setVolume(preMuteVolume);
      setMusicVolume(preMuteMusicVolume);
    }
  }

  /**
   * Check is music is currently playing.
   *
   * @return true if playing, else false
   */
  public boolean musicIsPlaying() {
    return music.isPlaying();
  }

  /**
   * Get game sound volume.
   *
   * @return game volume
   */
  public float getVolume() {
    return volume;
  }

  /**
   * Set game sound volume.
   *
   * @param volume volume value between 0 and 1
   */
  public void setVolume(float volume) {
    this.volume = Math.max(0.0f, Math.min(1.0f, volume)); // Clamp between 0.0 and 1.0
    if (windId != 0) {
      wind.setVolume(windId, this.volume);
    }
  }

  /**
   * Get music volume.
   *
   * @return music volume between 0 and 1
   */
  public float getMusicVolume() {
    return musicVolume;
  }

  /**
   * Set music volume.
   *
   * @param volume value between 0 and 1
   */
  public void setMusicVolume(float volume) {
    this.musicVolume = Math.max(0.0f, Math.min(1.0f, volume));
    music.setVolume(this.musicVolume);
  }
}
