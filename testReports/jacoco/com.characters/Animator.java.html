package com.characters;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.Animation;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import java.util.HashMap;

/** Generates animations from sprite sheets. */
public class Animator {

  private final int columns;
  private final int rows;
  private final Texture sheet;

  /** Cuts sprite sheets up into animations. */
  public HashMap<String, Animation<TextureRegion>> animations =
      new HashMap<String, Animation<TextureRegion>>();

  /**
   * Loads sprite sheet.
   *
   * @param file path of sprite sheet
   * @param sheetCols Number of columns in sheet
   * @param sheetRows Number of rows in sheet
   */
  public Animator(String file, int sheetCols, int sheetRows) {
    columns = sheetCols;
    rows = sheetRows;
    sheet = new Texture(Gdx.files.internal(file));
  }

  /**
   * Cuts the sprite sheet up Stores resulting animations into animations dictionary.
   *
   * @param animInfo dictionary of sprite sheet layout: key - Name of animation Value - Array
   *     representing row of animation on sprite sheet and index of start and end frames
   * @param animationSpeed number of seconds each animation frame should last
   */
  protected void generateAnimation(HashMap<String, Integer[]> animInfo, float animationSpeed) {
    // animationInfo [0] = Row of animation, [1] = start frame [2] = end frame

    TextureRegion[][] tmp =
        TextureRegion.split(sheet, sheet.getWidth() / columns, sheet.getHeight() / rows);

    TextureRegion[] frames;
    for (String key : animInfo.keySet()) {
      frames = new TextureRegion[animInfo.get(key)[2] - animInfo.get(key)[1]];
      for (int i = animInfo.get(key)[1]; i < animInfo.get(key)[2]; i++) {
        frames[i - animInfo.get(key)[1]] = tmp[animInfo.get(key)[0]][i];
      }
      animations.put(key, new Animation<TextureRegion>(animationSpeed, frames));
    }
  }

  /** Dispose of resources. */
  public void dispose() {
    if (sheet != null) {
      sheet.dispose();
    }
  }
}
