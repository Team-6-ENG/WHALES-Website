package com.characters;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Input;
import com.badlogic.gdx.graphics.Texture;
import com.badlogic.gdx.graphics.g2d.Sprite;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.graphics.g2d.TextureRegion;
import com.badlogic.gdx.math.Vector2;
import java.util.HashMap;

/** Players character manager. */
public final class Player extends Animator {
  // TODO: This should likely extend Actor

  public final Energy energy;

  /** The speed that the player moves by default. TODO: In units/second */
  public final float baseSpeed = 8f;

  // Changed: store path in constant for testing
  private final String image;
  private final Sprite playerSprite;

  /** The speed that the player moves when sprinting. TODO: In units/second */
  private final float sprintSpeedBonus = 8f;

  public TextureRegion currentPlayerFrame;

  /** The speed that the player moves. TODO: In units/second */
  private float speed = baseSpeed;

  /** Rotation of the player sprite. */
  private float rotation = 0f;

  private float woozyTimer = 0f;
  private boolean isSprint = false;
  private int verticalMovement = 0;
  private int horizontalMovement = 0;

  // Changed 17/12/25 to extend new animator class (all animation logic added)

  /**
   * Player constructor.
   *
   * @param sheetPath file path of sprite sheet
   * @param imgPath file path of sprite img
   */
  public Player(String sheetPath, String imgPath) {
    super(sheetPath, 7, 8);
    image = imgPath;
    HashMap<String, Integer[]> animationInfo = new HashMap<String, Integer[]>();
    // HashMap<String, Integer[]> animationInfo:
    //      key - Name of animation
    //      Value - Array representing row of animation on sprite sheet and index of start and end
    // frames
    animationInfo.put("walkForwards", new Integer[] {0, 0, 7});
    animationInfo.put("walkLeftForwards", new Integer[] {1, 0, 7});
    animationInfo.put("walkRightForwards", new Integer[] {6, 0, 7});
    animationInfo.put("walkLeft", new Integer[] {2, 0, 7});
    animationInfo.put("walkRight", new Integer[] {7, 0, 7});
    animationInfo.put("walkRightBackwards", new Integer[] {5, 0, 7});
    animationInfo.put("walkLeftBackwards", new Integer[] {3, 0, 7});
    animationInfo.put("walkBackwards", new Integer[] {4, 0, 7});

    generateAnimation(animationInfo, 0.3f);

    playerSprite = new Sprite(animations.get("walkLeftForwards").getKeyFrame(0, true));

    playerSprite.setOrigin(1, 1);
    this.energy = new Energy(50f);
  }

  /** Moves the player based on user input. */
  public void handleInput(int windy, float delta, float randomX, float randomY) {
    // The time since the last frame. Used to scale movement speed to keep it constant regardless of
    // framerate.
    Vector2 velocity = new Vector2(0, 0);

    verticalMovement = 0;
    horizontalMovement = 0;
    // Directional movement
    if (Gdx.input.isKeyPressed(Input.Keys.RIGHT) || Gdx.input.isKeyPressed(Input.Keys.D)) {
      velocity.x += 1;
      velocity.x += 1;
      horizontalMovement = 1;
    } else if (Gdx.input.isKeyPressed(Input.Keys.LEFT) || Gdx.input.isKeyPressed(Input.Keys.A)) {
      velocity.x -= 1;
      horizontalMovement = -1;
    }
    if (Gdx.input.isKeyPressed(Input.Keys.UP) || Gdx.input.isKeyPressed(Input.Keys.W)) {
      velocity.y += 1;
      verticalMovement = 1;
    } else if (Gdx.input.isKeyPressed(Input.Keys.DOWN) || Gdx.input.isKeyPressed(Input.Keys.S)) {
      velocity.y -= 1;
      verticalMovement = -1;
    }
    // Sprint
    if (Gdx.input.isKeyPressed(Input.Keys.SHIFT_LEFT)
        && energy.isAccessible()
        && !velocity.isZero()
        && !isSprint) {
      setSpeed(speed + sprintSpeedBonus);
      energy.reduceEnergy(delta * 4);
      energy.regenerating = false;
      isSprint = true;
    } else {
      if (isSprint) {
        setSpeed(speed - sprintSpeedBonus);
        isSprint = false;
      }
      if (!Gdx.input.isKeyPressed(Input.Keys.SHIFT_LEFT)) {
        energy.regenerating = true;
      }
    }
    // Normalise the vector so that diagonal movement isn't sqrt(2) times faster.
    velocity.nor();
    // Scale the vector according to speed
    velocity.scl(speed * delta);

    // Changed 10/12/25 to add wind resistance
    if (horizontalMovement == 0) {
      velocity.x += (windy * 0.05f);
    } else {
      velocity.x += (windy * 0.13f);
    }

    // changed 28/12/25 to add woozy event
    if (woozyTimer > 0) {
      woozyTimer -= delta;
      if (!velocity.isZero()) {
        velocity.x += randomX;
        velocity.y += randomY;
      }
      if (woozyTimer <= 0) {
        resetSpeed();
      }
    }
    move(velocity);
  }

  /**
   * Update player animation Added 17/12/25 to control new animations.
   *
   * @param stateTime time in seconds since last frame
   */
  public void updatePlayer(float stateTime) {
    if (verticalMovement == 1) {
      switch (horizontalMovement) {
        case 0:
          currentPlayerFrame = animations.get("walkBackwards").getKeyFrame(stateTime, true);
          break;
        case 1:
          currentPlayerFrame = animations.get("walkRightBackwards").getKeyFrame(stateTime, true);
          break;
        case -1:
          currentPlayerFrame = animations.get("walkLeftBackwards").getKeyFrame(stateTime, true);
          break;
        default:
          throw new IllegalStateException("Unexpected value: " + horizontalMovement);
      }
    } else if (verticalMovement == -1) {
      switch (horizontalMovement) {
        case 0:
          currentPlayerFrame = animations.get("walkForwards").getKeyFrame(stateTime, true);
          break;
        case 1:
          currentPlayerFrame = animations.get("walkRightForwards").getKeyFrame(stateTime, true);
          break;
        case -1:
          currentPlayerFrame = animations.get("walkLeftForwards").getKeyFrame(stateTime, true);
          break;
        default:
          throw new IllegalStateException("Unexpected value: " + horizontalMovement);
      }
    } else if (verticalMovement == 0) {
      switch (horizontalMovement) {
        case 0:
          currentPlayerFrame = new TextureRegion(new Texture(Gdx.files.internal(image)));
          break;
        case 1:
          currentPlayerFrame = animations.get("walkRight").getKeyFrame(stateTime, true);
          break;
        case -1:
          currentPlayerFrame = animations.get("walkLeft").getKeyFrame(stateTime, true);
          break;
        default:
          throw new IllegalStateException("Unexpected value: " + horizontalMovement);
      }
    }

    playerSprite.setSize(1.3f, 2.7f);
    playerSprite.setRegion(currentPlayerFrame);
  }

  /** Call after drinking too many energy drinks, makes player woozy. */
  public void sugarCrash() {
    woozyTimer = 10;
    setSpeed(2);
  }

  /**
   * Render the player to the screen.
   *
   * @param batch The SpriteBatch to render to
   */
  public void render(SpriteBatch batch) {
    playerSprite.setRotation(
        rotation); // rotation is rendered so must be checked on every render call
    playerSprite.draw(batch);
  }

  /**
   * Get player sprite.
   *
   * @return The player sprite
   */
  public Sprite getSprite() {
    return playerSprite;
  }

  /**
   * Get player speed.
   *
   * @return The current speed of the player
   */
  public float getSpeed() {
    return speed;
  }

  /**
   * Set the speed of the player.
   *
   * @param speed The new value of speed
   */
  public void setSpeed(float speed) {
    this.speed = speed;
  }

  /** Reset speed to `BASE_SPEED`. */
  public void resetSpeed() {
    this.speed = baseSpeed;
    if (isSprint) {
      this.speed += sprintSpeedBonus;
    }
  }

  /**
   * Get player position.
   *
   * @return The player sprite position
   */
  public Vector2 getPosition() {
    Vector2 position = new Vector2();
    return playerSprite.getBoundingRectangle().getPosition(position);
  }

  /**
   * Set the position of the player sprite.
   *
   * @param position The new position for the sprite.
   */
  public void setPosition(Vector2 position) {
    setPosition(position.x, position.y);
  }

  /**
   * Set the position of the player sprite.
   *
   * @param x The new x-position for the sprite.
   * @param y The new y-position for the sprite.
   */
  public void setPosition(float x, float y) {
    playerSprite.setPosition(x, y);
  }

  /**
   * Get player rotation.
   *
   * @return rotation multiplier
   */
  public float getRotation() {
    return rotation;
  }

  /**
   * Set rotation of the player.
   *
   * @param rotation rotation multiplier
   */
  public void setRotation(float rotation) {
    this.rotation = rotation;
  }

  /**
   * Move the player sprite.
   *
   * @param x The amount to move the sprite on the x-axis.
   * @param y The amount to move the sprite on the y-axis
   */
  public void move(float x, float y) {
    playerSprite.translate(x, y);
  }

  /**
   * Move the player sprite.
   *
   * @param vector The vector to move the player by
   */
  public void move(Vector2 vector) {
    move(vector.x, vector.y);
  }

  /** Releases all resources of this object. */
  public void dispose() {
    // Nothing to dispose currently.
  }
}
