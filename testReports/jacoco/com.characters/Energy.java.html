package com.characters;

import static com.badlogic.mazegame.Main.labelStyle;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.graphics.g2d.SpriteBatch;
import com.badlogic.gdx.scenes.scene2d.ui.Label;

/** Represents the energy which allows the player to sprint. */
public class Energy {
  private static final String labelFormatString = "Energy:%02d/%d";

  /** The maximum energy. */
  public final float maxEnergy;

  /** The rate at which energy is regenerated (energy/second). */
  public final float regenerationRate = 1;

  private final Label label;

  /** Whether the energy is currently regenerating. */
  public boolean regenerating = true;

  /** The current energy level. */
  private float currentEnergy;

  /** Whether energy can be used currently. */
  private boolean accessible = true;

  private float energyTickTimer;

  /**
   * Create a new energy bar.
   *
   * @param maxEnergy The maximum energy
   */
  Energy(float maxEnergy) {
    this.maxEnergy = maxEnergy;
    currentEnergy = maxEnergy / 2f;

    label = createLabel();
  }

  /**
   * Initialise text label. Changed 06/12/25 Moved out of constructor for testing
   *
   * @return text label
   */
  protected Label createLabel() {
    Label l =
        new Label(
            String.format(labelFormatString, Math.round(currentEnergy), Math.round(maxEnergy)),
            labelStyle);
    l.setPosition(0.5f, -4);
    l.setFontScale(0.03f);

    return l;
  }

  /**
   * Get amount of energy remaining.
   *
   * @return <code>currentEnergy</code>
   */
  public float getCurrentEnergy() {
    return currentEnergy;
  }

  /**
   * Reduce energy by <code>amount</code>.
   *
   * @param amount The amount to reduce the energy by.
   */
  public void reduceEnergy(float amount) {
    if (amount < 0) {
      throw new IllegalArgumentException(
          "Cannot reduce by a negative amount, use `increaseEnergy` instead");
    }
    currentEnergy = Math.max(currentEnergy - amount, 0);
    label.setText(
        String.format(labelFormatString, Math.round(currentEnergy), Math.round(maxEnergy)));

    // Block access to energy if it gets entirely depleted, until it's fully replenished again.
    if (currentEnergy == 0f) {
      accessible = false;
    }
  }

  /**
   * Increase energy by <code>amount</code>.
   *
   * @param amount The amount to reduce the energy by.
   */
  public void increaseEnergy(float amount) {
    if (amount < 0) {
      throw new IllegalArgumentException(
          "Cannot increase by a negative amount, use `decreaseEnergy` instead");
    }
    currentEnergy = Math.min(currentEnergy + amount, maxEnergy);
    label.setText(
        String.format(labelFormatString, Math.round(currentEnergy), Math.round(maxEnergy)));

    // Re-enable access to energy when it's fully restored
    if (currentEnergy >= 1) {
      accessible = true;
    }
  }

  /**
   * Check if energy can be used.
   *
   * @return Whether energy can be used currently
   */
  public boolean isAccessible() {
    return accessible;
  }

  /** Update the energy, regenerating it based on delta time. */
  public void tick(float delta) {
    // Regenerate energy if enabled.
    if (regenerating) {
      // Update timer
      energyTickTimer += delta;
      // Timer ran out, reset and regenerate energy
      if (energyTickTimer > 1) {
        energyTickTimer = 0;
        increaseEnergy(regenerationRate);
      }
    }
  }

  /** Render the timer label. */
  public void render(SpriteBatch batch) {
    label.draw(batch, 255);
  }
}
